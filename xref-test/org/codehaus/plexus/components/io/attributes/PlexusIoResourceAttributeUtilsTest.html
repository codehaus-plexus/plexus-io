217     public void testReversedMonthDayOrder()
218         throws Exception
219     {
220         String output = //
221             "-rw-r--r--   1 501  80  7683 31 May 10:06 pom_newer.xml\n" + //
222             "-rwxr--r--   1 502  81  7683  1 Jun 2010  pom_older.xml";
223         InputStream byteArrayInputStream = new ByteArrayInputStream( output.getBytes() );
224         NumericUserIDAttributeParser parser = getNumericParser1nla();
225         parse( byteArrayInputStream, parser );
226         Map<String, PlexusIoResourceAttributes> map = parser.getAttributesByPath();
227         
228         // 6 months or newer ls date format
229         FileAttributes newerFileAttr = (FileAttributes) map.get( "pom_newer.xml" );
230         assertNotNull( newerFileAttr );
231         assertEquals( "-rw-r--r--", new String( newerFileAttr.getLsModeParts() ) );
232         assertEquals( 501, newerFileAttr.getUserId().intValue() );
233         assertEquals( 80, newerFileAttr.getGroupId().intValue() );
234         
235         // older than 6 months ls date format
236         FileAttributes olderFileAttr = (FileAttributes) map.get( "pom_older.xml" );
237         assertNotNull( olderFileAttr );
238         assertEquals( "-rwxr--r--", new String( olderFileAttr.getLsModeParts() ) );
239         assertEquals( 502, olderFileAttr.getUserId().intValue() );
240         assertEquals( 81, olderFileAttr.getGroupId().intValue() );

591     private void parse( InputStream stream, AttributeParser parser )
592         throws Exception
593     {
594         BufferedReader bufferedReader = new BufferedReader( new InputStreamReader( stream ) );
595         String line = bufferedReader.readLine();
596         int lineNum = 0;
597         try
598         {
599             while ( line != null )
600             {
601                 parser.consumeLine( line );
602                 line = bufferedReader.readLine();
603                 lineNum++;
604             }
605         }
606         catch ( Exception e )
607         {
608             Exception exception = new Exception( "At line " + lineNum + "in source:" + line );
609             exception.initCause( e );
610             throw e;
611         }

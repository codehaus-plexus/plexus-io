18  @SuppressWarnings("OctalInteger")
19  class AttributeUtilsTest {
20      @Test
21      void miscPatterns() {
22          final Set<PosixFilePermission> permissions = AttributeUtils.getPermissions(0124);
23          assertTrue(permissions.contains(PosixFilePermission.OWNER_EXECUTE));
24          assertTrue(permissions.contains(PosixFilePermission.GROUP_WRITE));
25          assertTrue(permissions.contains(PosixFilePermission.OTHERS_READ));
26      }
27  
28      @Test
29      void morePatterns() {
30          final Set<PosixFilePermission> permissions = AttributeUtils.getPermissions(0241);
31          assertTrue(permissions.contains(PosixFilePermission.OWNER_WRITE));
32          assertTrue(permissions.contains(PosixFilePermission.GROUP_READ));
33          assertTrue(permissions.contains(PosixFilePermission.OTHERS_EXECUTE));
34      }
35  
36      @Test
37      void evenMorePatterns() {
38          final Set<PosixFilePermission> permissions = AttributeUtils.getPermissions(0412);
39          assertTrue(permissions.contains(PosixFilePermission.OWNER_READ));
40          assertTrue(permissions.contains(PosixFilePermission.GROUP_EXECUTE));
41          assertTrue(permissions.contains(PosixFilePermission.OTHERS_WRITE));
42      }
43  
44      @Test
45      void test777() {
46          final Set<PosixFilePermission> permissions = AttributeUtils.getPermissions(0777);
47          assertEquals(9, permissions.size());
48      }
49  
50      @Test
51      @DisabledOnOs(OS.WINDOWS)
52      void chmodBackAndForth() throws Exception {
53          final File bxx = File.createTempFile("bxx", "ff");
54          AttributeUtils.chmod(bxx, 0422);
55          PlexusIoResourceAttributes firstAttrs = new FileAttributes(bxx);
56          assertTrue(firstAttrs.isOwnerReadable());
57          assertFalse(firstAttrs.isOwnerWritable());
58          assertFalse(firstAttrs.isOwnerExecutable());
59          AttributeUtils.chmod(bxx, 0777);
60          PlexusIoResourceAttributes secondAttrs = new FileAttributes(bxx);
61          assertTrue(secondAttrs.isOwnerReadable());
62          assertTrue(secondAttrs.isOwnerWritable());
63          assertTrue(secondAttrs.isOwnerExecutable());
64      }

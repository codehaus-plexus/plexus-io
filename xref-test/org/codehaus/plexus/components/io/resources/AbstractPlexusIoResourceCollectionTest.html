17  public class AbstractPlexusIoResourceCollectionTest
18      extends TestCase
19  {
20      @SuppressWarnings( "ResultOfMethodCallIgnored" ) public void testGetIncludes()
21          throws Exception
22      {
23          AbstractPlexusIoResourceCollection sut = new AbstractPlexusIoResourceCollection()
24          {
25              public Iterator<PlexusIoResource> getResources()
26                  throws IOException
27              {
28                  return Arrays.asList(getResource( "r1" ), getResource( "r2" )).iterator();
29              }
30  
31              public Stream stream()
32              {
33                  throw new UnsupportedOperationException();
34              }
35  
36          };
37  
38          sut.setStreamTransformer( new InputStreamTransformer()
39          {
40              @Nonnull
41              public InputStream transform( @Nonnull PlexusIoResource resource, @Nonnull final InputStream inputStream )
42                  throws IOException
43              {
44                  final byte[] buf = new byte[2];
45                  buf[0] = (byte) inputStream.read();
46                  buf[1] = (byte) inputStream.read();
47                  return new ByteArrayInputStream( buf );
48              }
49          } );
50  
51          final PlexusIoResource next = sut.getResources().next();
52          final InputStream inputStream = sut.getInputStream( next );
53          inputStream.read();
54          inputStream.read();
55          assertEquals( -1, inputStream.read());
56          inputStream.close();
57  
58      }
59  
60      private static PlexusIoResource getResource( final String r1 )
61      {
62          return new AbstractPlexusIoResource( r1, 0, 0, true, false, true )
63          {
64              @Nonnull
65              public InputStream getContents()
66                  throws IOException
67              {
68                  return new ByteArrayInputStream( (r1 + "Payload").getBytes() );
69              }
70  
71              public URL getURL()
72                  throws IOException
73              {
74                  throw new IllegalStateException( "Not implemented" );
75              }
76          };
77      }

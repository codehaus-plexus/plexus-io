17  /**
18   * Tests for PlexusIoFileResource with DeferredFileOutputStream scenarios.
19   * This specifically tests the fix for issue #118 where temp files were deleted
20   * before they could be accessed, causing FileNotFoundException.
21   */
22  class PlexusIoFileResourceDeferredTest {
23  
24      @TempDir
25      File tempDir;
26  
27      @Test
28      void fileResourceWithTransformerCanReadContentsMultipleTimes() throws Exception {
29          // Create a test file with content larger than typical buffer size
30          File testFile = new File(tempDir, "test-file.txt");
31          byte[] largeContent = new byte[10000]; // 10KB
32          for (int i = 0; i < largeContent.length; i++) {
33              largeContent[i] = (byte) ('A' + (i % 26));
34          }
35          Files.write(testFile.toPath(), largeContent);
36  
37          // Create a transformer that modifies the content
38          InputStreamTransformer transformer = (resource, inputStream) -> {
39              // Simple transformer that reads and returns the same content
40              return inputStream;
41          };
42  
43          // Create PlexusIoFileResource with transformer
44          PlexusIoFileResource resource =
45                  new PlexusIoFileResource(testFile, testFile.getName(), new FileAttributes(testFile), null, transformer);
46  
47          // First read - this should work and not delete the temp file
48          try (InputStream is1 = resource.getContents()) {
49              byte[] read1 = IOUtils.toByteArray(is1);
50              assertEquals(largeContent.length, read1.length);
51          }
52  
53          // Second read - this should also work if the temp file wasn't prematurely deleted
54          // This is the key test - without the fix, this would throw FileNotFoundException
55          try (InputStream is2 = resource.getContents()) {
56              byte[] read2 = IOUtils.toByteArray(is2);
57              assertEquals(largeContent.length, read2.length);
58          }
59      }
60  
61      @Test
62      void fileResourceWithTransformerLargeFile() throws Exception {
63          // Create a large file that exceeds the DeferredFileOutputStream threshold (5MB)
64          File testFile = new File(tempDir, "large-test-file.bin");
65          byte[] chunk = new byte[1024 * 1024]; // 1MB chunks
66          for (int i = 0; i < chunk.length; i++) {
67              chunk[i] = (byte) i;
68          }
69  
70          // Write 6MB to exceed the 5MB threshold
71          try (OutputStream os = Files.newOutputStream(testFile.toPath())) {
72              for (int i = 0; i < 6; i++) {
73                  os.write(chunk);
74              }
75          }
76  
77          InputStreamTransformer transformer = (resource, inputStream) -> inputStream;
78  
79          PlexusIoFileResource resource =
80                  new PlexusIoFileResource(testFile, testFile.getName(), new FileAttributes(testFile), null, transformer);
81  
82          // Verify we can read the content - this tests that the temp file
83          // created by DeferredFileOutputStream is properly accessible
84          long size = resource.getSize();
85          assertTrue(size > 5_000_000, "File should be larger than 5MB threshold");
86  
87          try (InputStream is = resource.getContents()) {
88              assertNotNull(is);
89              byte[] firstBytes = new byte[1024];
90              int read = is.read(firstBytes);
91              assertEquals(1024, read);
92          }
93      }
94  
95      @Test
96      void fileResourceWithTransformerSmallFile() throws Exception {
97          // Test with a small file that stays in memory (below 5MB threshold)
98          File testFile = new File(tempDir, "small-test-file.txt");
99          String content = "Hello, World!";
100         Files.write(testFile.toPath(), content.getBytes(StandardCharsets.UTF_8));
101 
102         InputStreamTransformer transformer = (resource, inputStream) -> inputStream;
103 
104         PlexusIoFileResource resource =
105                 new PlexusIoFileResource(testFile, testFile.getName(), new FileAttributes(testFile), null, transformer);
106 
107         // Multiple reads should work for small files too
108         for (int i = 0; i < 3; i++) {
109             try (InputStream is = resource.getContents()) {
110                 String readContent = IOUtils.toString(is, StandardCharsets.UTF_8);
111                 assertEquals(content, readContent);
112             }
113         }
114     }

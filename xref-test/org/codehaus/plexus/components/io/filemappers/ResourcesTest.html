54      private File getTestDir()
55      {
56          final String testDirPath = System.getProperty( "plexus.io.testDirPath" );
57          return new File( testDirPath == null ? "target/plexus.io.testDir" : testDirPath );
58      }

62          return new File( getTestDir(), "files" );

65      private void createFiles()
66          throws IOException
67      {
68          final File baseDir = getFilesDir();
69          FileUtils.deleteDirectory( baseDir );
70          FileUtils.mkdir( baseDir.getPath() );
71          final File aFile = new File( baseDir, A_PATH );
72          FileUtils.mkdir( aFile.getParentFile().getPath() );
73          FileOutputStream fos = new FileOutputStream( aFile );
74          fos.write( "0123456789".getBytes( "US-ASCII" ) );
75          fos.close();
76          final File bFile = new File( baseDir, B_PATH );
77          fos = new FileOutputStream( bFile );
78          fos.write( "abcdefghijklmnopqrstuvwxyz".getBytes( "US-ASCII" ) );
79          fos.close();
80          final File yDir = new File( baseDir, Y_PATH );
81          FileUtils.mkdir( yDir.getPath() );
82      }

84      private void addDirToZipFile( ZipOutputStream zos, File dir, String path )
85          throws IOException
86      {
87          final String prefix = path == null ? "" : ( path + "/" );
88          File[] files = dir.listFiles();
89          for ( File f : files )
90          {
91              final String entryName = prefix + f.getName();
92              ZipEntry ze = new ZipEntry( entryName );
93              if ( f.isFile() )
94              {
95                  ze.setSize( f.length() );
96                  zos.putNextEntry( ze );
97                  FileInputStream fis = new FileInputStream( f );
98                  byte[] buffer = new byte[1024];
99                  for ( ;; )
100                 {
101                     int res = fis.read( buffer );
102                     if ( res == -1 )
103                     {
104                         break;
105                     }
106                     if ( res > 0 )
107                     {
108                         zos.write( buffer, 0, res );
109                     }
110                 }
111                 fis.close();
112                 ze.setTime( f.lastModified() );
113                 zos.closeEntry();
114             }
115             else
116             {
117                 addDirToZipFile( zos, f, entryName );
118             }
119         }
120     }

122     private void createZipFile( File dest, File dir ) throws IOException
123     {
124         FileOutputStream fos = new FileOutputStream( dest );
125         ZipOutputStream zos = new ZipOutputStream( fos );
126         addDirToZipFile( zos, dir, null );
127         zos.close();
128     }

130     private void compare( InputStream in, File file )
131         throws IOException
132     {
133         try( InputStream fIn = new FileInputStream( file ) )
134         {
135             for ( ;; )
136             {
137                 int i1 = in.read();
138                 int i2 = fIn.read();
139                 assertEquals( i2, i1 );
140                 if ( i1 == -1 )
141                 {
142                     break;
143                 }
144             }
145         }
146     }

148     private void compare( PlexusIoResource res, File file )
149         throws IOException
150     {
151         assertTrue( res.getLastModified() != PlexusIoResource.UNKNOWN_MODIFICATION_DATE );
152         if ( res instanceof PlexusIoFileResource )
153         {
154             assertEquals( res.getLastModified() / 1000, file.lastModified() / 1000 );
155         }
156         assertTrue( res.getSize() != PlexusIoResource.UNKNOWN_RESOURCE_SIZE );
157         assertEquals( res.getSize(), file.length() );
158         InputStream in = res.getContents();
159         compare( in, file );
160         in.close();
161         URLConnection uc = res.getURL().openConnection();
162         uc.setUseCaches( false );
163         in = uc.getInputStream();
164         compare( in, file );
165         in.close();
166     }

168     private void testPlexusIoResourceCollection( PlexusIoResourceCollection plexusIoResourceCollection )
169         throws IOException
170     {
171         boolean xPathSeen = false;
172         boolean yPathSeen = false;
173         boolean aFileSeen = false;
174         boolean bFileSeen = false;
175         Iterator iter = plexusIoResourceCollection.getResources();
176         while ( iter.hasNext() )
177         {
178             PlexusIoResource res = (PlexusIoResource) iter.next();
179             final String resName = res.getName().replace( File.separatorChar, '/' );
180             if ( res.isDirectory() )
181             {
182                 assertFalse( "The directory " + resName + " is a file.", res.isFile() );
183                 if ( X_PATH.equals( resName ) )
184                 {
185                     xPathSeen = true;
186                 }
187                 else if ( Y_PATH.equals( resName ) )
188                 {
189                     yPathSeen = true;
190                 }
191                 else if ( "".equals( resName ) || ".".equals( resName ) )
192                 {
193                     // Ignore me
194                 }
195                 else
196                 {
197                     fail( "Unexpected directory entry: " + resName );
198                 }
199                 final File dir = new File( getFilesDir(), resName );
200                 assertTrue( "The directory " + dir + " doesn't exist.", dir.isDirectory() );
201             }
202             else
203             {
204                 assertTrue( "The file " + resName + " isn't reported to be a file.", res.isFile() );
205                 assertTrue( "The file " + resName + " doesn't exist.", res.isExisting() );
206                 final File f = new File( getFilesDir(), resName );
207                 assertTrue( "A file " + f + " doesn't exist.", f.isFile() && f.exists() );
208                 if ( A_PATH.equals( resName ) )
209                 {
210                     aFileSeen = true;
211                 }
212                 else if ( B_PATH.equals( resName ) )
213                 {
214                     bFileSeen = true;
215                 }
216                 else
217                 {
218                     fail( "Unexpected file entry: " + resName );
219                 }
220                 compare( res, f );
221             }
222         }
223 
224         assertTrue( aFileSeen );
225         assertTrue( bFileSeen );
226         if ( iter instanceof Closeable )
227         {
228             ( (Closeable) iter ).close();
229         }
230     }

233         throws IOException
234     {
235         resourceCollection.setBaseDir( getFilesDir() );
236         testPlexusIoResourceCollection( resourceCollection );

31      protected void testFileMapper( FileMapper pMapper, String[] pInput, String[] pOutput )
32      {
33          for ( int i = 0; i < pInput.length; i++ )
34          {
35              final String input = pInput[i];
36              final String output = pOutput[i];
37              final String result;
38              try
39              {
40                  if ( output == null )
41                  {
42                      try
43                      {
44                          pMapper.getMappedFileName( input );
45                          fail( "Expected IllegalArgumentException for mapper " + pMapper.getClass().getName()
46                                          + " and input: " + input );
47                      }
48                      catch ( IllegalArgumentException e )
49                      {
50                          // Ok
51                      }
52                      continue;
53                  }
54                  result = pMapper.getMappedFileName( input );
55                  if ( output.equals( result ) )
56                  {
57                      continue;
58                  }
59              }
60              catch ( Throwable t )
61              {
62                  throw new UndeclaredThrowableException( t, "Mapper " + pMapper.getClass().getName()
63                                  + " failed for input " + input + ": " + t.getMessage() );
64              }
65              if ( !output.equals( result ) )
66              {
67                  fail( "Mapper " + pMapper.getClass().getName() + " failed for input nr. " + i
68                  		+ ", " + input + ": Expected " + output
69                          + ", got " + result );
70              }
71          }
72      }

86      {

102     @Test
103     public void testFileExtensionMapper() throws Exception
104     {
105         final String[] results = getIdentityResults();
106         for ( int i = 2; i <= 10; i += 2 )
107         {
108             results[i] += ".png";
109         }
110         for ( int i = 3; i <= 11; i += 2 )
111         {
112             results[i] = results[i].substring( 0, results[i].length() - ".gif".length() ) + ".png";
113         }
114         testFileExtensionMapper( results, new FileExtensionMapper() );
115         testFileExtensionMapper( results, (FileExtensionMapper) lookup( FileMapper.class, FileExtensionMapper.ROLE_HINT ) );
116     }

119     {

121         testFileMapper( mapper, SAMPLES, results );

202     @Test
203     public void testRegExpFileMapper() throws Exception
204     {
205         final String[] results = getIdentityResults();
206         results[3] = "xyz.jpg";
207         results[5] = "b/xyz.jpg";
208         results[7] = "b\\xyz.jpg";
209         results[9] = "c.c/xyz.jpg";
210         results[11] = "c.c\\xyz.jpg";
211         testFileMapper( configure(new RegExpFileMapper(), "\\.gif$", ".jpg"), SAMPLES, results );
212         testFileMapper( configure(new RegExpFileMapper(), "^(.*)\\.gif$", "$1.jpg"), SAMPLES, results );
213         testFileMapper( configure((RegExpFileMapper) lookup( FileMapper.class, RegExpFileMapper.ROLE_HINT ), "\\.gif$", ".jpg"), SAMPLES, results );
214         final RegExpFileMapper mapper = configure( new RegExpFileMapper(), "c", "f" );
215         mapper.setReplaceAll( true );
216         final String[] fResults = getIdentityResults();
217         fResults[8] = "f.f/a";
218         fResults[9] = "f.f/xyz.gif";
219         fResults[10] = "f.f\\a";
220         fResults[11] = "f.f\\xyz.gif";
221         testFileMapper( mapper, SAMPLES, fResults );
222     }

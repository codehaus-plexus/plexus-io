19  import java.io.IOException;
20  import java.io.InputStream;
21  import java.util.Iterator;
22  
23  /**
24   * A resource collection is a set of {@link PlexusIoResource} instances.
25   */
26  public interface PlexusIoResourceCollection extends Iterable<PlexusIoResource>
27  {
28      /**
29       * Returns an iterator over the resources in the collection.
30       * @return An iterator
31       * @throws java.io.IOException .
32       */
33      Iterator<PlexusIoResource> getResources() throws IOException;
34  
35      /**
36       * Returns the resources as a stream.
37       * @return A stream for functional iteration
38       */
39      Stream stream();
40  
41      /**
42       * Returns the resources suggested name. This is used for
43       * integrating file mappers.
44       * @param resource A resource, which has been obtained by
45       *   calling {@link #getResources()}.
46       * @return The resource name. If it is a file, it should be normalized to platform separators
47       */
48      String getName( PlexusIoResource resource );
49  
50      /**
51       * Returns the collections last modification time. For a
52       * collection of files, this might be the last modification
53       * time of the file, which has been modified at last. For an
54       * archive file, this might be the modification time of the
55       * archive file.
56       * @return {@link PlexusIoResource#UNKNOWN_MODIFICATION_DATE},
57       *   if the collections last modification time is unknown,
58       *   otherwise the last modification time in milliseconds.
59       * @throws java.io.IOException .
60       */
61      long getLastModified() throws IOException;
62  
63      /**
64       * Returns an input stream for the provided resource, with stream transformers applied
65       * @param resource The resources
66       * @return A possibly transformed resource
67       * @throws IOException when something goes bad
68       */
69      InputStream getInputStream( PlexusIoResource resource ) throws IOException;
70  
71      /**
72       * Resolves the supplied resource into a "real" resource. Resolving
73       * means applying input transformations
74       * Returns an input stream for the provided resource, with stream transformers applied
75       * @param resource The resources
76       * @return A possibly transformed resource
77       * @throws IOException when something goes bad
78       */
79      PlexusIoResource resolve( PlexusIoResource resource ) throws IOException;
80  
81      /**
82       * Indicates if this collection supports concurrent access to its resources.
83       * <p>Some resource collections (like tar files) may not support efficient random access
84       * or seek operation so implementations that represent such collections may not be able
85       * to provide concurrent access to its resources. If implementation returns {@code false},
86       * then it is not safe to access its methods and resources in concurrent fashion.
87       * For example it is not safe to read from two resources in two concurrent threads,
88       * to read a resource and iterate over the iterator returned by {@link #getResources()}
89       * in two concurrent threads, etc.
90       * <p>Please note that this method indicates concurrent support only for the collection,
91       * not for the individual resources. This means there is no guarantee that
92       * the resources returned by {@link #resolve(PlexusIoResource)} or the input stream
93       * returned by {@link #getInputStream(PlexusIoResource)} are thread-safe,
94       * even if {@code true} is returned.
95       * @return {@code true} if this collection supports concurrent access,
96       *   otherwise {@code false}
97       */
98      boolean isConcurrentAccessSupported();
99  
100 }
